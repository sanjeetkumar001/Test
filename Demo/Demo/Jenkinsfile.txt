pipeline{
	agent any
		stages{
			stage('one'){
				steps{
					echo 'Hi,This Declarative Pipeline Demo'
				}
			}
			stage('Two'){
				steps{
					input('Do you want to proceed?')
				}
			}
			stage('Three'){
				when {
					not {
						branch "master"
						}
					}
					step {
						echo "Hello"
					}
				}
			stage('Four'){
					parallel
						stage('unit Test') {
								steps {
									echo "Running the unit test..."
								}
					}
					stage('Integration test'){
							agent{
								docker {
									reuseNode false
									image 'ubuntu'
								}
							}
							steps{
								echo 'Running the integration test..'
							}
					}
				}
}
}
}
}
